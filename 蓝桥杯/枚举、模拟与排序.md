# 枚举、模拟与排序
## 归并排序
1. 找到分界点
2. 递归排序 `left` `right` 
3. ==合并==
O(nlogn)
### 标准归并
```cpp
#include <iostream>

using namespace std;

const int N = 100010;
int n;
int q[N], temp[N];

void merge_sort(int q[], int l, int r) {
    if (l >= r) return;
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);
    
    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r) {
        if (q[i] <= q[j]) temp[k ++] = q[i ++];
        else temp[k ++] = q[ j ++];
    }
    while (i <= mid) temp[k ++] = q[i ++];
    while (j <= r) temp[k ++] = q[ j ++];
    
    for (i = l, j = 0; i <= r; i ++, j ++) q[i] = temp[j];
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i ++) scanf("%d", &q[i]);
    merge_sort(q, 0, n - 1);
    for (int i = 0; i < n; i ++) printf("%d ", q[i]);
    return 0;
}
```

```cpp
#include <iostream>
using namespace std;

const int N = 100010;
typedef long long LL;
int n; 
int q[N], temp[N];

LL merge_sort(int l, int r) {
    if (l >= r) return 0;
    int mid = l + r >> 1;
    LL res = merge_sort(l, mid) + merge_sort(mid + 1, r);
    
    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r) {
        if (q[j] < q[i]) {
            res += mid - i + 1;
            temp[k ++] = q[j ++];
        } else temp[k ++] = q[i ++];
    }
    while (i <= mid) temp[k ++] = q[i ++];
    while (j <= r) temp[k ++] = q[j ++];
    
    for (int i = l, j = 0; i <= r; i ++, j ++) q[i] = temp[j];
    return res;
}

int main() {
    cin >> n;
    for (int i = 0; i < n; i ++) cin >> q[i];
    cout << merge_sort(0, n - 1) << endl;
    
    return 0;
}
```
